DICIONÁRIO DE DADOS

Tabela Nome da coluna Tipo de dados Comprimento Restrições Descrição
TB_USUARIO ID_USER INT --- PK, NOT NULL, AUTOMATIC numeração automática
NOME_USER VARCHAR 60 BYTES NOT NULL nome que o usuário terá
SENHA_USER VARCHAR 20 BYTES NOT NULL senha que o usuário escolherá
STATUS_USER VARCHAR 30 BYTES NOT NULL definição se o usuário está ativo ou inativo
TB_VENDEDOR ID_VENDEDOR INT --- PK, NOT NULL numeração automática
NOME_VENDEDOR VARCHAR 60 BYTES NOT NULL nome do vendedor
FONE_VENDEDOR CHAR 16 BYTES NOT NULL telefone para contato do vendedor
EMAIL_VENDEDOR VARCHAR 60 BYTES NOT NULL email do vendedor que deverá conter @
SENHA_VENDEDOR VARCHAR 20 BYTES NOT NULL senha para o login do vendedor
CPF_VENDEDOR CHAR 14 BYTES PK, NOT NULL cadastro do CPF unico e validado
DTA_CAD_VENDEDOR DATE --- NOT NULL data e hora do cadastro
DTA_NASC_VENDEDOR DATE --- NOT NULL data de nascimento
STATUS_VENDEDOR VARCHAR 1 BYTE NOT NULL status se o vendedor é 1. ativo ou 0.inativo
IMAGEM_VENDEDOR VARCHAR 200 BYTES NOT NULL link para a imagem do vendedor
ID_USER INT --- FK, NOT NULL ID da tabela TB_USUARIO
TB_CLIENTE ID_CLIENTE INT --- PK, NOT NULL numeração automática
NOME_CLIENTE VARCHAR 60 BYTES NOT NULL nome do cliente
FONE_CLIENTE CHAR 16 BYTES NOT NULL telefone para contato do consumidor
EMAIL_CLIENTE VARCHAR 60 BYTES NOT NULL email do consumidor que deverá conter @
SENHA_CLIENTE VARCHAR 20 BYTES NOT NULL senha para o login do consumidor
CPF_CLIENTE CHAR 14 BYTES PK, NOT NULL cadastro do CPF unico e validado
DTA_CAD_CLIENTE DATE --- NOT NULL data e hora do cadastro
DTA_NASC_CLIENTE DATE --- NOT NULL data de nascimento
STATUS_CLIENTE VARCHAR 40 BYTES NOT NULL status se o comprador é 1. ativo ou 0.inativo
IMAGEM_CLIENTE VARCHAR 200 BYTES NOT NULL link para a imagem do consumidor
ID_USER INT --- FK, NOT NULL ID da tabela TB_USUARIO
TB_ENDERECO ID_END INT --- PK, NOT NULL numeração automática
LOGRADOURO_END VARCHAR 60 BYTES NOT NULL logradouro do usuário
CEP_END CHAR 8 BYTES NOT NULL CEP do usuário
NUM_END VARCHAR 7 BYTES NOT NULL número e/ou letra do bloco da moradia
COMPLEMENTO_END VARCHAR 50 BYTES NOT NULL complemento do endereço
ID_USER INT --- FK, NOT NULL ID da tabela TB_USUARIO
TB_CHAT ID_CHAT INT --- PK, NOT NULL numeração automática
MENSAGEM_CHAT VARCHAR 800 BYTES NOT NULL mensagem de troca entre o vendedor e consumidor
TIME_ENVIO_CHAT DATE --- NOT NULL horário em que a mensagem foi enviada no chat
DES_CHAT VARCHAR 40 BYTES NOT NULL Descrição do Chat
ID_ANUNCIO INT --- FK, NOT NULL ID da tabela TB_ANUNCIO
ID_CLIENTE INT --- FK, NOT NULL ID da tabela TB_CLIENTE
TB_ANUNCIO ID_ANUNCIO INT --- PK, NOT NULL numeração automática
NOME_ANUNCIO VARCHAR 80 BYTES NOT NULL Nome do produto anunciado
DESC_ANUNCIO VARCHAR 200 BYTES NOT NULL Descrição do produto anunciado
VALOR_VENDA_ANUNCIO DEC (7,2) BYTES NOT NULL valor do produto
STATUS_ANUNCIO VARCHAR 40 BYTES NOT NULL Status se está disponivel, ja foi vendido etc
QTD_ANUNCIO INT 3 BYTES NOT NULL Quantidade disponivel do produto, caso tenha mais
ID_VENDEDOR INT --- FK, NOT NULL ID da tabela TB_VENDEDOR
TB_VENDA ID_VENDA INT --- PK, NOT NULL numeração automática
DTA_VENDA DATE --- NOT NULL Data e hora da venda
VALOR_VENDA DEC (7,2) BYTES NOT NULL Valor que o produto foi vendido
STATUS_VENDA VARCHAR 40 BYTES NOT NULL status da venda se ela foi efetuada, revogada etc
ID_ANUNCIO INT --- FK, NOT NULL ID da tabela TB_ANUNCIO
ID_CLIENTE INT --- FK, NOT NULL ID da tabela TB_CLIENTE
